## 配置项

```ts
{
  /**
   * 刷新阈值 达到该值松手即可刷新
   * @default 50
   */
  refreshThreshold?: number
  /**
   * 最大下拉距离
   * @default 100
   */
  maxPullDistance?: number
  /**
   * 下拉角度,角度大于该值则不触发下拉
   * @default 30
   */
  pullAngle?: number
  /**
   * 加载器大小
   * @default 40
   */
  size?: number
  /**
   * 刷新中回调 返回Promise true则关闭刷新
   * @returns
   */
  onRefresh: () => Promise<boolean>
  /**
   * 局部滚动与滚动元素事件绑定
   */
  event?: EventManager<React.TouchEvent | React.PointerEvent>
  onChange?: () => void
  className?: string
  style?: React.CSSProperties
}
```
